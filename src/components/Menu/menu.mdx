> Menu菜单组件

使用方式如下所示：

import { Playground, Props } from 'docz'
import {Menu,MenuItem,SubMenu} from './index.tsx'


<Playground>
    <Menu
    defaultIndex="0"
    defaultOpenSubMenus={[]}
    mode="horizontal"
  >
    <MenuItem>
      cool link
    </MenuItem>
    <MenuItem>
      cool link 2
    </MenuItem>
    <MenuItem disabled>
      disabled
    </MenuItem>
    <SubMenu title="下拉选项">
      <MenuItem>
        下拉选项一
      </MenuItem>
      <MenuItem>
        下拉选项二
      </MenuItem>
    </SubMenu>
  </Menu>
</Playground>
